\begin{prooftree}
\AxiomC{$x : T \in \Gamma$}
\LeftLabel{\textsc{T-Var}: }
\UnaryInfC{$\Gamma \vdash x : T$}
\end{prooftree}
\begin{prooftree}
\AxiomC{$\Gamma \vdash t : \mathcal{F}\, ( T_0 \ldots T_n ) \, T$}
\AxiomC{$\Gamma \vdash t_0 : \mathcal{F}\, T_0$}
\AxiomC{$\dotsc$}
\AxiomC{$\Gamma \vdash t_n : \mathcal{F}\, T_n$}
\LeftLabel{\textsc{T-App}: }
\QuaternaryInfC{$\Gamma \vdash t \, ( t_0 \ldots t_n ) : \mathcal{F}\, T$}
\end{prooftree}
\begin{prooftree}
\AxiomC{$\Gamma \vdash t : T$}
\AxiomC{$T \sub T'$}
\LeftLabel{\textsc{T-Sub}: }
\BinaryInfC{$\Gamma \vdash t : T'$}
\end{prooftree}
\begin{prooftree}
\AxiomC{}
\LeftLabel{\textsc{S-Refl}: }
\UnaryInfC{$T <: T$}
\end{prooftree}
\begin{prooftree}
\AxiomC{$T \sub T'$}
\AxiomC{$T' \sub T''$}
\LeftLabel{\textsc{S-Trans}: }
\BinaryInfC{$T \sub T''$}
\end{prooftree}
\begin{prooftree}
\AxiomC{}
\LeftLabel{\textsc{S-Const}: }
\UnaryInfC{$T \sub ( S_0 \ldots S_m )\, T$}
\end{prooftree}
\begin{prooftree}
\AxiomC{$T \sub T'$}
\AxiomC{$T_0 \sub T_0'$}
\AxiomC{$\dotsc$}
\AxiomC{$T_n \sub T_n'$}
\LeftLabel{\textsc{S-Arrow}: }
\QuaternaryInfC{$
  (T_0' \ldots T_n' )\, T
\sub
  (T_0 \ldots T_n )\, T'
$}
\end{prooftree}

\begin{prooftree}
\AxiomC{}
\LeftLabel{\textsc{SA-Atom}: }
\UnaryInfC{$\vdash_\mathrm{A} A_i <: A_i$}
\end{prooftree}
\begin{prooftree}
\AxiomC{$\vdash_\mathrm{A} T \sub T'$}
\LeftLabel{\textsc{SA-Const}: }
\UnaryInfC{$\vdash_\mathrm{A} T \sub ( S_0 \ldots S_m )\, T'$}
\end{prooftree}
\begin{prooftree}
\AxiomC{$\vdash_\mathrm{A} T \sub T'$}
\AxiomC{$\vdash_\mathrm{A} T_0 \sub T_0'$}
\AxiomC{$\dotsc$}
\AxiomC{$\vdash_\mathrm{A} T_n \sub T_n'$}
\LeftLabel{\textsc{SA-Arrow}: }
\QuaternaryInfC{$
\vdash_\mathrm{A}
  (T_0' \ldots T_n')\, T
\sub
  (T_0 \ldots T_n)\, T'
$}
\end{prooftree}

このとき$T \sub T' \iff \vdash_\mathrm{A} T \sub T'$（たぶん）．
