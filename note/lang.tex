\begin{center}
\begin{tabular}{rcll}
$T$ &$\to$& $\mathrm{A}$ & （原子型） \\

\end{tabular}

\begin{prooftree}
\AxiomC{$x : T \in \Gamma$}
\LeftLabel{\textsc{T-Var}: }
\UnaryInfC{$\Gamma \vdash x : T$}
\end{prooftree}
\begin{prooftree}
\AxiomC{$\Gamma \vdash t : \texttt{(} T_0 \texttt{,} \dotsc \texttt{,} T_n \texttt{):} T$}
\AxiomC{$\Gamma \vdash t_0 : T_0$}
\AxiomC{$\dotsc$}
\AxiomC{$\Gamma \vdash t_n : T_n$}
\LeftLabel{\textsc{T-App0}: }
\QuaternaryInfC{$\Gamma \vdash t \texttt{(} t_0 \texttt{,} \dotsc \texttt{,} t_n \texttt{)} : T$}
\end{prooftree}
\begin{prooftree}
\AxiomC{$\Gamma \vdash t : \texttt{(} T_0 \texttt{,} \dotsc \texttt{,} T_n \texttt{):} T$}
\AxiomC{$\Gamma \vdash t_0 : \texttt{(} S_0 \texttt{,} \dotsc \texttt{,} S_m \texttt{):} T_0$}
\AxiomC{$\dotsc$}
\AxiomC{$\Gamma \vdash t_n : \texttt{(} S_0 \texttt{,} \dotsc \texttt{,} S_m \texttt{):} T_n$}
\LeftLabel{\textsc{T-App1}: }
\QuaternaryInfC{$\Gamma \vdash t \texttt{(} t_0 \texttt{,} \dotsc \texttt{,} t_n \texttt{)} : \texttt{(}S_0\texttt{,}\dotsc\texttt{,}S_m\texttt{):} T$}
\end{prooftree}
\begin{prooftree}
\AxiomC{$\Gamma \vdash t : T$}
\AxiomC{$T \sub T'$}
\LeftLabel{\textsc{T-Sub}: }
\BinaryInfC{$\Gamma \vdash t : T'$}
\end{prooftree}
\begin{prooftree}
\AxiomC{}
\LeftLabel{\textsc{S-Refl}: }
\UnaryInfC{$T <: T$}
\end{prooftree}
\begin{prooftree}
\AxiomC{$T \sub T'$}
\AxiomC{$T' \sub T''$}
\LeftLabel{\textsc{S-Trans}: }
\BinaryInfC{$T \sub T''$}
\end{prooftree}
\begin{prooftree}
\AxiomC{}
\LeftLabel{\textsc{S-Const}: }
\UnaryInfC{$T \sub \texttt{(} S_0 \texttt{,} \ldots \texttt{,} S_m \texttt{):} T$}
\end{prooftree}
\begin{prooftree}
\AxiomC{$T \sub T'$}
\AxiomC{$T_0 \sub T_0'$}
\AxiomC{$\dotsc$}
\AxiomC{$T_n \sub T_n'$}
\LeftLabel{\textsc{S-Arrow}: }
\QuaternaryInfC{$
  \texttt{(}T_0'\texttt{,}\ldots\texttt{,}T_n'\texttt{):}T
\sub
  \texttt{(}T_0\texttt{,}\ldots\texttt{,}T_n\texttt{):}T'
$}
\end{prooftree}

\begin{prooftree}
\AxiomC{}
\LeftLabel{\textsc{SA-Refl}: }
\UnaryInfC{$\vdash_\mathrm{A} T <: T$}
\end{prooftree}
\begin{prooftree}
\AxiomC{$\vdash_\mathrm{A} T \sub T'$}
\LeftLabel{\textsc{SA-Const}: }
\UnaryInfC{$\vdash_\mathrm{A} T \sub \texttt{(} S_0 \texttt{,} \ldots \texttt{,} S_m \texttt{):} T'$}
\end{prooftree}
\begin{prooftree}
\AxiomC{$\vdash_\mathrm{A} T \sub T'$}
\AxiomC{$\vdash_\mathrm{A} T_0 \sub T_0'$}
\AxiomC{$\dotsc$}
\AxiomC{$\vdash_\mathrm{A} T_n \sub T_n'$}
\LeftLabel{\textsc{SA-Arrow}: }
\QuaternaryInfC{$
\vdash_\mathrm{A}
  \texttt{(}T_0'\texttt{,}\ldots\texttt{,}T_n'\texttt{):}T
\sub
  \texttt{(}T_0'\texttt{,}\ldots\texttt{,}T_n'\texttt{):}T'
$}
\end{prooftree}

このとき$T \sub T' \iff \vdash_\mathrm{A} T \sub T'$（たぶん）．
